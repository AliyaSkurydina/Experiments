\section {Введение}
\begin{frame}
\frametitle{Введение}
%\fontsize{8pt}{9}\selectfont
Одними из важнейших задач являются обратные задачи гравиметрии и магнитометрии о восстановлении поверхностей раздела слоев в однородно-слоистой среде. Задачи возникают при поиске полезных ископаемых и построении геолого-геофизических моделей. Они описываются интегральными уравнениями первого рода и являются некорректными задачами: имеют неединственное решение, неустойчиво зависящее от входных данных. 
\\
Исследованию задач о восстановлении поверхностей посвящены работы В.~Н.~Страхова, А.~В.~Цирульского, П.~С.~Мартышко, И.~Л.~Пруткина, В.~М.~Новоселицкого, В.~Б.~Гласко, Е.~Г.~Булаха, А.~И.~Кобрунова, В.~А.~Кочнева, П.~А.~Миненко и др.
\\
\end{frame}

\begin{frame}
\frametitle{1.1.1 Обратная задача гравиметрии}
Гравитационный эффект от поверхности раздела:
\begin{equation}\nonumber
\begin{aligned}
&f  \Delta \sigma \int \limits_{-\infty}^{\infty} \int \limits_{-\infty}^{\infty} \Bigg [ \frac{1}{\big((x-x_0)^2+(y-y_0)^2+\textcolor{blue}{\zeta^2(x_0,y_0)}\big)^{1/2}} - \\
&- \frac{1}{\big((x-x_0)^2+(y-y_0)^2+h^2\big)^{1/2}}\Bigg ] dx_0 dy_0 = \Delta g(x,y,0)
\end{aligned}
\end{equation}
Это уравнение является нелинейным интегральным уравнением первого рода от $\zeta$:\footnotemark 
\begin{equation}\nonumber
\begin{aligned}
A(\zeta) = \Delta g(x,y,0).
\end{aligned}
\end{equation}

\end{frame}

\begin{aligned}

Исключив слои, ограниченные горизонтальными плоскостями и вносящие нулевой вклад в магнитное поле, сведем прямую задачу к задаче нахождения поля от субгоризонтального слоя между $z_1(x,y)=\zeta(x,y)$ и $z_2(x,y)=h$. \\
По теореме Пуассона магнитный потенциал $U$ однородного намагниченного тела находится по формуле\\
\begin{equation} \nonumber
\begin{aligned}
&U=- \frac{1}{4\pi} \langle \boldsymbol{J},\nabla V \rangle, \\
\end{aligned}
\end{equation}
\fontsize{9pt}{11}\selectfont
где $V$ --- гравитационный потенциал тела такой же формы, который можно найти по формуле
\begin{equation} \nonumber
\begin{aligned}
&V(\boldsymbol{r}_0)= f  \sigma \int\limits_D \frac {1}{ | \boldsymbol{r}-\boldsymbol{r}_0 |} dr.
\end{aligned}
\end{equation}
\end{columns}
\end{frame}

\begin{block}{}

\begin{equation}\nonumber
\begin{aligned}
\Delta Z(x,y,0) & = - \frac{1}{4\pi} \int \limits_{-\infty}^{\infty} \int \limits_{-\infty}^{\infty} \Bigg [ \frac{\Delta J_x(x_0-x)+ \Delta J_y(y_0-y)-\Delta J_z h}{\big((x-x_0)^2+(y-y_0)^2+h^2\big)^{3/2}} -\\
& - \frac{\Delta J_x(x_0-x)+\Delta J_y(y_0-y)-\Delta J_z \textcolor{blue}{\zeta(x_0,y_0)} }{\big((x-x_0)^2+(y-y_0)^2+\textcolor{blue}{\zeta^2(x_0,y_0)}\big)^{3/2}} \Bigg ] dx_0 dy_0,
\end{aligned}
\end{equation}
где $\Delta \boldsymbol{J}=\boldsymbol{J}_2-\boldsymbol{J}_1$ ($\boldsymbol{J}=\{J_x, J_y, J_z\}$) - разность намагниченностей.

\subsection{Линеаризованный метод сопряженных градиентов}

\begin{frame}
\fontsize{9pt}{7.2}\selectfont
\frametitle{2.1 Построение быстрых алгоритмов на основе метода сопряженных градиентов}
Получен из варианта нелинейного метода сопряженных градиентов для корректных задач оптимизации (Р.~Флетчер, Б.~Т.~Поляк, Х.~Носедаль) с~добавлением демпфирующего параметра, впервые введенного В.~В.~Васиным. \\
Рассматривается система вида
\begin{equation}\nonumber
A[z]=F
\end{equation}
\fontsize{9pt}{7.2}\selectfont
В общем виде нелинейный метод сопряженных градиентов имеет следующий вид
\begin{equation}\nonumber
\begin{aligned}
& z^{k+1} = z^k - \chi^k p^k, \\
& p^k=S(z^k)+\beta^k p^{k-1}, \quad p^0=S(z^0), \\
& S(z^k)=A'(z^k)^T(A(z^k)-F).\\
\end{aligned}
\end{equation}
\fontsize{9pt}{7.2}\selectfont
Для параметра $\beta$ применим формулу Полака-Рибьера-Поляка 
\begin{equation} \nonumber
\beta^k=\max \left \{ \frac{<S(z^k),(S(z^k)-S(z^{k-1}))>}{\left \| S(z^{k-1})\right \|^2},0 \right \}
\end{equation}
%\fontsize{8pt}{7.2}\selectfont
\let\thefootnote\relax\let\thefootnote\relax\footnotetext{\footnotesize Vasin V. V., Eremin I. I. Operators and iterative processes of Fejer type: theory and applications. Walter de Gruyter, 2009. Vol. 53.}
\end{frame}

\begin{frame}
%\fontsize{8pt}{7.2}\selectfont
\frametitle{2.1 Получение явной приближенной формулы для размера шага}
В классических методах размер шага $\chi$ находится на каждой итерации из задачи одномерной минимизации
\begin{equation}\nonumber
\chi^k=\argmin\limits_{\chi} {\left \{ \left \|A(z^k-\chi p^k)-F \right \|^2\right \} }.
\end{equation}
Заменим эту процедуру \textcolor{blue}{явной формулой} с использованием линейного приближения $A(z)\cong A(z^k)+A'(z^k)(z-z^k)$:
\begin{equation} \nonumber
\widetilde{\chi_k} = \argmin\limits_{\chi} \Big \{\left \|A(z^k)+A'(z^k)(z^k+\chi p^k-z^k)-F \right \|^2 \Big \}.
\end{equation}
Получим выражение для размера шага
\begin{equation} \nonumber
\textcolor{blue}{\chi_k \approx \widetilde{\chi_k}= -\frac{\langle p^k, S(z^k) \rangle}{\left \|A'(z^k) p^k \right \|^2}.}
\end{equation}

\end{frame}

\begin{frame}
\fontsize{8pt}{7.2}\selectfont
\frametitle{2.1 Линеаризованный метод сопряженных градиентов}
Метод ЛМСГ приобретет следующий вид 
\fontsize{12pt}{7.2}\selectfont
\begin{equation}\nonumber
\begin{aligned}
& z^{k+1} = z^k - \textcolor{blue}{\psi \frac{<p^k, S(z^k)>}{\left \| A'(z^k)p^k\right \|^2}}p^k, \\\nonumber
& p^k=S(z^k)+\beta^k p^{k-1}, \quad p^0=S(z^0), \\\nonumber
& \beta^k=\max \left \{ \frac{<S(z^k),(S(z^k)-S(z^{k-1}))>}{\left \| S(z^{k-1})\right \|^2},0 \right \},\\\nonumber
& S(z^k)=A'(z^k)^T(A(z^k)-F).
\end{aligned}
\end{equation}
\fontsize{8pt}{7.2}\selectfont

Условие останова: $\left \| A(z) - F \right \|/\left \| F \right \| < \varepsilon$ при достаточно малом $\varepsilon$, начальное приближение $z^0 \equiv H$ либо априорно известная поверхность. \\
\vspace{10pt}
Демпфирующий параметр $\psi \leqslant 1$ выбирается экспериментально: максимальный, при котором уменьшается норма невязки.

\fontsize{8pt}{7.2}\selectfont
\let\thefootnote\relax\let\thefootnote\relax\footnotetext{\footnotesize Акимова, Е. Н., Мисилов, В. Е., Скурыдина, А. Ф., Третьяков, А. И. Градиентные методы решения структурных обратных задач гравиметрии и магнитометрии на суперкомпьютере “Уран” // Вычислительные методы и программирование, 2015. Т. 16, 155–164.}
%\let\thefootnote\relax\let\thefootnote\relax\footnotetext{\smallEngl, Heinz Werner, Martin Hanke, and Andreas Neubauer. Regularization of inverse problems. Vol. 375. Springer Science \& Business Media, 1996.}.
\end{frame}

\begin{frame}
\frametitle{2.1 Регуляризованный вариант метода}
%\begin{block}{}
\fontsize{8pt}{7.2}\selectfont
Применим к задаче регуляризацию по методу Тихонова:
\begin{equation} \nonumber
S_{\alpha}(z)=A'(z)^T(A(z)-F)+\textcolor{blue}{\alpha (z-z^0)}.
\end{equation}
Получим регуляризованный линеаризованный метод сопряженных градиентов (РЛМСГ):
\begin{equation}\nonumber
\begin{aligned}
& z^{k+1} = z^k - \psi \frac{ \langle p^k, S_{\alpha}(z^k) \rangle}{\left \| A'(z^k)p^k\right \|^2 + \textcolor{blue}{\alpha \left \| p^k \right \|^2}}p^k, \quad p^k=S_{\alpha}(z^k)+\beta^k p^{k-1},\\
& p^0=S_{\alpha}(z^0), \quad \beta^k=\max \left \{ \frac{\langle S_{\alpha}(z^k),(S_{\alpha}(z^k)-S_{\alpha}(z^{k-1}))\rangle }{\left \| S_{\alpha}(z^{k-1})\right \|^2},0 \right \},%\\
%& S(z)=A'(z)^T(A(z)-F)+\textcolor{blue}{\alpha(z-z^0)}.
\end{aligned}
\end{equation}
\fontsize{8pt}{7.2}\selectfont
где $z^0$ --- начальное приближение, $\alpha$ --- параметр регуляризации.
%\end{block}\footnotemark
\let\thefootnote\relax\let\thefootnote\relax\footnotetext{\footnotesize Akimova E.N., Misilov V.E., Tretyakov A.I. Regularized methods for solving nonlinear inverse gravity problem // EAGE Geoinformatics 2016 – 15th Intern. Conference on Geoinformatics – Theoretical and Applied Aspects}
\end{frame}

\subsection {Покомпонентный градиентный метод}

\begin{frame}
\frametitle{2.2 Построение \textcolor{blue}{нового} экономичного покомпонентного градиентного метода}
\fontsize{9pt}{11}\selectfont
Возьмем одно из уравнений системы: $A_i(z)=F_i$. Задача нахождения из него $z$ эквивалентна задаче минимизации $\min\left \{\frac{1}{2} ( A_i(z)-F_i  )^2:z \in R^n\right \}$. Построим для ее решения промежуточный процесс градиентного типа:
\begin{equation}\nonumber
z^{k+1}={z}^{k}+ \chi  (A_i(z^k)-F_i) \nabla A_i(z^k)^T.
\end{equation}
Размер шага $\chi^k$ находится из задачи одномерной минимизации $\chi^k=\argmin\limits_{\chi^k}\{(A_i(z^{k+1})-F_i)^2\}.$
%\begin{equation}\nonumber
%\alpha^k=\argmin\limits_{\alpha^k}\{(A_i(z^{k+1})-F_i)^2\}.
%\end{equation}
Воспользовавшись линейным приближением $A_i(z)\simeq A_i(z^k)+\nabla A_i(z^k)(z-z^k)$, получим 
\begin{equation}\nonumber
\chi_k\approx -\frac{1}{\left \| \nabla A_i(z^k) \right \|^2}.
\end{equation}
Воспользуемся следующей идеей: оставим из процесса только ту часть, которая отвечает за получение $i$-го элемента $z^{k+1}_i$ вектора приближенного решения на следующей итерации. %Кроме того, введем демпфирующий параметр $\psi$.

\begin{frame}
\frametitle{2.3 Новый алгоритм решения обратной задачи для случая многослойной среды}
\begin{block}{}
Новый алгоритм решения интегральных уравнений позволяет по суммарному полю из уравнения находить несколько структурных границ одновременно. Алгоритм основан на применении градиентных методов с переменными весовыми множителями. \\

\textcolor{blue}{Идея: используются весовые коэффициенты} для каждой из поверхностей и для каждого узла сетки по отдельности. Это обеспечивает более тонкую настройку, и дает возможность за счет выбора весовых множителей отсеивать нефизичные решения, что снижает неопределенность.

%Весовые множители выбираются с использованием полей, выделенных из наблюденных данных по методике повысотных трансформаций либо с использованием начальных приближений поверхностей.\\
\end{block}
\begin{block}{Преимущества алгоритма}
Как показали численные эксперименты, применение нового алгоритма существенно сокращает число итераций и время счета.
\end{block}

\begin{frame}
\frametitle{2.3.1 Способ дискретизации уравнений}
После дискретизации правой части на сетке $n=MxN$ и аппроксимации интегрального оператора по квадратурным формулам прямоугольников имеем векторы правой части $F$ размерности $n$ и решения $z$ размерности $Ln$. 
\begin{equation}\nonumber
\begin{aligned}
&f \sum \limits_{l=1}^{L} \Delta \sigma_l \Delta x \Delta y \sum \limits_{j=1}^{Ln}  \Bigg [ \frac{1}{\big((x-x_0)^2+(y-y_0)^2+z_j^2\big)^{1/2}} - \\
&- \frac{1}{\big((x-x_0)^2+(y-y_0)^2+H_l^2\big)^{1/2}}\Bigg ] = F,
\end{aligned}
\end{equation}

\begin{equation}\nonumber
\begin{aligned}
&\sum \limits_{l=1}^{L} \frac{\Delta J_l}{4\pi} \Delta x \Delta y \sum \limits_{j=1}^{Ln} \Bigg [ \frac{z_j}{\big((x-x_0)^2+(y-y_0)^2+z_j^2\big)^{3/2}} -\\
& - \frac{H_l}{\big((x-x_0)^2+(y-y_0)^2+H_l^2\big)^{3/2}} \Bigg ] = F.
\end{aligned}
\end{equation}
\end{frame}

\begin{frame}
\frametitle{2.3.2 Выбор весовых множителей}
Весовые множители $\gamma_i$ для каждой компоненты $z_i(i=1..Ln)$ можно выбирать двумя способами:
\begin{itemize}
\item используя начальные приближения (если известны)
\begin{equation} \nonumber
\begin{aligned}
\hat{Z} &= [\hat{Z_1},\hat{Z_2},...,\hat{Z_L}]=(\hat{z}_1,\hat{z}_2,...,\hat{z}_{Ln}) \longrightarrow(\gamma_1,\gamma_2,...,\gamma_{Ln}), \\
\hat{Z_l} &\longrightarrow (\gamma_{(l-1)n+1},\gamma_{(l-1)n+2},...,\gamma_{ln)}), \quad \textcolor{blue}{\gamma_i=\frac{|\nicefrac{\hat{z_i}}{H_l}-1|^\beta}{\max \limits_{i}|\nicefrac{\hat{z_i}}{H_l}-1|^\beta},}
\end{aligned}
\end{equation}
\item используя выделенные поля
\begin{equation} \nonumber
\begin{aligned}
F &\longrightarrow [F_1, F_2,...,F_L]=(f_1,f_2,...,f_{Ln}) \longrightarrow(\gamma_1,\gamma_2,...,\gamma_{ln}), \\
F_l &\longrightarrow (\gamma_{(l-1)n+1},\gamma_{(l-1)n+2},...,\gamma_{ln}), \quad \textcolor{blue}{\gamma_i=\frac{|f_i|^\beta}{\max \limits_{i}|f_i|^\beta},}
\end{aligned}
\end{equation}
\end{itemize}
где $\beta>1$ — параметр сглаживания.
\end{frame}


\begin{frame}
\frametitle{2.3.3 Градиентные методы с переменными весовыми множителями}
\fontsize{8pt}{9}\selectfont
%Модификация методов с постоянным множителем. 
%\scalebox{0.7} {
%\begin{minipage}{\textwidth}
\begin{block}{}
\begin{itemize}
\item линеаризованный метод наискорейшего спуска (ЛМНС)
\begin{equation} \nonumber
z^{k+1}=z^k - \textcolor{blue}{\gamma} \circ \frac{\left \| S(z^k) \right \|^2}{\left \|A'(z^k)S(z^k) \right \|^2} S(z^k);
\end{equation}
\item линеаризованный метод минимальной ошибки (ЛММО)
\begin{equation} \nonumber
z^{k+1}=z^k - \textcolor{blue}{\gamma} \circ \frac{\left \| A(z_k)-F \right \|^2}{\left \| S(z^k) \right \|^2} S(z^k),
\end{equation}
\end{itemize}
\begin{equation} \nonumber
S(z)=A'(z)^T(A(z)-F),
\end{equation}
где $k$ --- номер итерации, $\gamma=\{\gamma_1,\gamma_2,...,\gamma_{Ln}\}$ — вектор весовых множителей, $\circ$ --- покомпонентное перемножение векторов.
\end{block}\footnotemark

\begin{block}{}
\begin{itemize}
\item Модифицированный метод наискорейшего спуска (ЛММНС)
\begin{equation} \nonumber
z^{k+1}=z^k- \psi \frac {<\textcolor{blue}{v^k}, S(z^k)>}{\left \| A'(z^k) \textcolor{blue}{v^k} \right \|^2} \textcolor{blue}{v^k};
\end{equation}
\item Модифицированный метод сопряженных градиентов (ЛММСГ)
\begin{equation}\nonumber
\begin{aligned}

& \quad p^0=\textcolor{blue}{v^0},  \beta^k=\max \left \{ \frac{<\textcolor{blue}{v^k},(\textcolor{blue}{v^k}-\textcolor{blue}{v^{k-1}})>}{\left \| \textcolor{blue}{v^{k-1}}\right \|^2},0 \right \},\\
& S(z)=A'(z)^T(A(z)-F), \textcolor{blue}{v^k= \gamma \circ S(z^k)},
\end{aligned}
\end{equation}
\end{itemize}
где $z_i$ --- $i$-я компонента результирующего вектора $z$, $k$ --- номер итерации, $\gamma$ — весовые множители, $\psi$ --- демпфирующий параметр.
\textcolor{blue}{Преимущество — снижение числа итераций.}
